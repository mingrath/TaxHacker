    Add `getUpcomingDeadlines` function:
    ```typescript
    import { getDeadlinesForMonth } from "@/services/filing-deadlines"
    import { getFilingStatusesForMonth } from "@/models/filing-status"

    export type DeadlineWithStatus = {
      deadline: FilingDeadline
      status: "pending" | "filed" | "overdue"
      filedAt: Date | null
    }

    export async function getUpcomingDeadlines(userId: string): Promise<DeadlineWithStatus[]> {
      // Get current month and previous month deadlines (previous month deadlines may still be upcoming)
      const now = new Date()
      const currentMonth = now.getMonth() + 1
      const currentYear = now.getFullYear()

      // Get deadlines for previous month (filing is due in current month)
      const prevMonth = currentMonth === 1 ? 12 : currentMonth - 1
      const prevYear = currentMonth === 1 ? currentYear - 1 : currentYear
      const deadlines = getDeadlinesForMonth(prevMonth, prevYear)

      // Get filing statuses for these deadlines
      const statuses = await getFilingStatusesForMonth(userId, prevMonth, prevYear)
      const statusMap = new Map(statuses.map(s => [`${s.formType}-${s.taxMonth}-${s.taxYear}`, s]))

      // Map deadlines to status: filed if status exists and is "filed", overdue if past deadline, else pending
      return deadlines.map(d => {
        const key = `${d.formType}-${d.taxMonth}-${d.taxYear}`
        const filingStatus = statusMap.get(key)
        let status: "pending" | "filed" | "overdue" = "pending"
        if (filingStatus?.status === "filed") status = "filed"
        else if (d.adjustedDeadline < now) status = "overdue"
        return { deadline: d, status, filedAt: filingStatus?.filedAt ?? null }
      })
    }
    ```
