    4. **Create app/(app)/apps/tax-invoice/actions.ts** -- Server actions:

    Per D-05: Form-based creation with Section 86/4 enforcement. Sequential numbering auto-managed.
    Per D-06: Creating an invoice auto-creates the corresponding income transaction.

    ```typescript
    "use server"
    import { ActionState } from "@/lib/actions"
    import { getCurrentUser } from "@/lib/auth"
    import { prisma } from "@/lib/db"
    import { taxInvoiceFormSchema } from "@/forms/tax-invoice"
    import { getBusinessProfile } from "@/models/business-profile"
    import { getContactById } from "@/models/contacts"
    import { computeVATOnSubtotal } from "@/services/tax-calculator"

    export type TaxInvoiceData = {
      id: string
      documentNumber: string
      // ... all fields needed for PDF rendering
    }

    export async function createTaxInvoiceAction(
      prevState: ActionState<TaxInvoiceData> | null,
      formData: FormData
    ): Promise<ActionState<TaxInvoiceData>> {
      const user = await getCurrentUser()
      const parsed = taxInvoiceFormSchema.safeParse(/* parse formData */)
      if (!parsed.success) return { success: false, error: parsed.error.errors[0].message }

      // 1. Load business profile (seller info -- fields 2, 8)
      const profile = await getBusinessProfile(user.id)
      if (!profile.companyName || !profile.taxId) {
        return { success: false, error: "กรุณากรอกข้อมูลธุรกิจก่อน" }
      }

      // 2. Load contact (buyer info -- field 3)
      const contact = await getContactById(user.id, parsed.data.contactId)
      if (!contact) return { success: false, error: "ไม่พบผู้ซื้อ" }

      // 3. Compute totals
      const subtotal = parsed.data.items.reduce((sum, item) => sum + item.quantity * item.unitPrice, 0)
      const vatResult = computeVATOnSubtotal(subtotal)

      // 4. Generate sequential number (per INV-02)
      // Use Setting model with code "seq_tax_invoice" for atomic increment
      const nextNumber = await prisma.$transaction(async (tx) => {
        const setting = await tx.setting.upsert({
          where: { userId_code: { userId: user.id, code: "seq_tax_invoice" } },
          update: { value: { increment: 1 } },  // Note: may need raw SQL or custom logic
          create: { userId: user.id, code: "seq_tax_invoice", value: "1" },
        })
        return setting.value
      })
      // Format: INV-YYYY-NNNN (e.g., INV-2569-0001)

      // 5. Store invoice in AppData (or dedicated record)
      // Store the full invoice data for PDF regeneration

      // 6. Auto-create income Transaction (per D-06)
      await prisma.transaction.create({
        data: {
          userId: user.id,
          type: "income",
          name: `ใบกำกับภาษี ${documentNumber}`,
          merchant: contact.name,
          total: vatResult.total,
          subtotal: vatResult.subtotal,
          vatType: "output",
          vatAmount: vatResult.vatAmount,
          vatRate: 700,
          merchantTaxId: contact.taxId,
          merchantBranch: contact.branch,
          documentNumber,
          contactId: contact.id,
          issuedAt: new Date(parsed.data.issuedAt),
          currencyCode: "THB",
          items: JSON.stringify(parsed.data.items),
        }
      })

      return { success: true, data: invoiceData }
    }
    ```

    Handle the sequential numbering carefully: the Setting model stores `value` as String. Use a pattern like: read current value, parse as int, increment, save back -- all within `prisma.$transaction`. Or use AppData. The key requirement per INV-02 is strict monotonic increment with no gaps.
